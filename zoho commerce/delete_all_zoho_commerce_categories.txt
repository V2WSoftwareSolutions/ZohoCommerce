// Step 1: Get all categories

];
categories = response.get("categories");
// Step 2: Loop through and delete each category
for each  category in categories
{
	category_id = category.get("category_id");
	deleteResponse = invokeurl
	[
		url :"https://commerce.zoho.com/store/api/v1/categories/" + category_id + "?organization_id=" + organization_id
		type :DELETE
		connection:"full_access"
	];
	info "Deleted category: " + category_id;
	info deleteResponse;
}
