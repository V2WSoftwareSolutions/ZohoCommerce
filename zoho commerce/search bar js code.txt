<script>
(function () {
  const isHome = () =>
    (window.zs_resource_url === "/" || window.location.pathname === "/");

  function resizeSearchBar() {
    if (!isHome()) return;

    if (!document.getElementById("custom-homepage-style")) {
      const style = document.createElement("style");
      style.id = "custom-homepage-style";
      style.textContent = `
        /* ðŸŽ¨ Reduce search bar width and height on Home page */
        .theme-search.theme-search-and-mini-cart-group-item[data-zs-search-container] {
          max-width: 250px !important;
          height: 32px !important;
          border-radius: 18px !important;
          border: 1px solid #dcdcdc !important;
          background: #f8f8f8 !important;
        }
        .theme-search.theme-search-and-mini-cart-group-item input[type="search"] {
          height: 32px !important;
          font-size: 13px !important;
          padding: 4px 10px !important;
          border-radius: 18px !important;
        }
        .theme-search.theme-search-and-mini-cart-group-item input::placeholder {
          font-size: 13px !important;
          color: #888 !important;
        }
      `;
      document.head.appendChild(style);
      console.log("ðŸŽ¨ Search bar resized for home page");
    }
  }

  // Run when page loads
  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", resizeSearchBar);
  } else {
    resizeSearchBar();
  }

  // Keep applied if Zoho re-renders header
  new MutationObserver(resizeSearchBar)
    .observe(document.documentElement, { childList: true, subtree: true });
})();
</script>
